type CountControlsProps = {
suma: number;
resta: number;
reset: number;
incrementar: number;
decrementar: number;
}

function Contador(props: CountControlsProps) {
  const [count, setCount] = useState(0);

  function incrementar() {
    setCount(count + props.incrementar);
  }
  function suma() {
    setCount(count + props.suma);
  }

  function decrementar() {
    setCount(count - props.decrementar);
  }
  function resta() {
    setCount(count - props.resta);
  }

  function reset() {
    setCount(0);
  }

  return (
    <div>
      <h2>Contador: {count}</h2>
      <button onClick={incrementar}>+10</button>
      <button onClick={suma}>+1</button>
      <button onClick={decrementar}>-10</button>
      <button onClick={resta}>-1</button>
      <button onClick={reset}>Resetear</button>
    </div>
  );
}
